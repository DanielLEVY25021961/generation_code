	 * {@inheritDoc}
	 */
	@Override
	public boolean exists(
			final Long pId) throws AbstractDaoException {
		
		/* retourne false si pId == null . */
		if (pId == null) {
			return false;
		}
		
		/* retourne true si l'objet m√©tier existe en base. */
		if (this.findById(pId) != null) {
			return true;
		}
		
		return false;
